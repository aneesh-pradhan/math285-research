\section*{Nomenclature}
\begin{IEEEdescription}[\IEEEusemathlabelsep\IEEEsetlabelwidth{$\Rcrit$}]
  \item[$R$] Series resistance, measured in \si{\ohm}.
  \item[$L$] Series inductance, measured in \si{\henry}.
  \item[$C$] Series capacitance, measured in \si{\farad}.
  \item[$V(t)$] Applied source voltage.
  \item[$\Vs$] Step-source amplitude.
  \item[$\vC(t)$] Capacitor voltage, the primary output variable.
  \item[$i(t)$] Series current through the resistor, inductor, and capacitor.
  \item[$\alpha$] Damping coefficient, $\alpha = R/(2L)$.
  \item[$\omega_0$] Undamped natural frequency, $\omega_0 = 1/\sqrt{LC}$.
  \item[$\omega_d$] Damped natural frequency for the underdamped case.
  \item[$\Rcrit$] Critical resistance, $\Rcrit = 2\sqrt{L/C}$.
\end{IEEEdescription}
