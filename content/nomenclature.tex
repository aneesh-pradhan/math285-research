\section*{Nomenclature}
\begin{IEEEdescription}[\IEEEusemathlabelsep\IEEEsetlabelwidth{$\Rcrit$}]
  \item[$R$] Series resistance, measured in $\Omega$.
  \item[$L$] Series inductance, measured in H.
  \item[$C$] Series capacitance, measured in F.
  \item[$V(t)$] Applied source voltage.
  \item[$\Vs$] Step-source amplitude.
  \item[$\vC(t)$] Capacitor voltage, the primary output variable.
  \item[$\dot{v}_C(t)$] Time derivative of capacitor voltage, $dv_C/dt$,
        measured in V/s.
  \item[$i(t)$] Series current through the resistor, inductor, and capacitor.
  \item[$\alpha$] Damping coefficient, $\alpha = R/(2L)$.
  \item[$\omega_0$] Undamped natural frequency, $\omega_0 = 1/\sqrt{LC}$.
  \item[$\omega_d$] Damped natural frequency for the underdamped case.
  \item[$\Rcrit$] Critical resistance, $\Rcrit = 2\sqrt{L/C}$.
  \item[$R_0$] Nominal resistance used in the non-ideal simulation.
  \item[$L_0$] Nominal inductance used in the non-ideal simulation.
  \item[$k$] Heating coefficient in the current-dependent resistance model,
        measured in $\Omega/\mathrm{A}^2$.
  \item[$I_{\mathrm{sat}}$] Saturation current threshold for the inductor model.
  \item[$y_1,y_2$] State variables used in the numerical simulation.
\end{IEEEdescription}
